# أوامر رفع المشروع على GitHub

# 1. تهيئة Git في مجلد المشروع
git init

# 2. إضافة جميع الملفات للتتبع
git add .

# 3. إنشاء أول commit
git commit -m "نظام محاسبة متقدم باللغة العربية مع دعم متعدد قواعد البيانات"

# 4. ربط المشروع بـ GitHub (استبدل USERNAME و REPOSITORY-NAME)
git remote add origin https://github.com/USERNAME/REPOSITORY-NAME.git

# 5. تعيين الفرع الرئيسي
git branch -M main

# 6. رفع المشروع
git push -u origin main
